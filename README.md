# Solicitar al usuario que ingrese un número

if numero == 0:
    print("El número es cero.")
# Verificar si el número es par
elif numero % 2 == 0:
    print("El número es par.")
# Si no es par ni cero, entonces es impar
else:
    print("El número es impar.")
